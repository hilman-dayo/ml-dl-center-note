  import matplotlib.pyplot as plt
  import numpy as np


  def step(x):
      return (x > 0).astype(np.int)


  def sigmoid(x):
      return 1.0 / (1 + np.exp(-x))


  x = np.linspace(-6, 6, 100)
  plt.plot(x, step(x), label="Step")
  plt.plot(x, sigmoid(x), label="Sigmoid")
  plt.legend()
  plt.show()

#+end_src
